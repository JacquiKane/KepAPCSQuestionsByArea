public static int format(Employee[] arr) {
   for(int index = 0; index < arr.length; index++) {
    int pos = index - 1;
     while (pos >= 0 && arr[pos].getNumber() > arr[index].getNumber()) {
       arr[pos + 1] = arr[pos];
       pos = pos - 1;
       }
       arr[pos + 1] = arr[index];
    }
    }
